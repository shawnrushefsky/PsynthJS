PsynthJS

PsynthJS is a Javascript library for programmatically generating graphs in Psymphonic Psynth. It requires jQuery.

<script src="PATH TO JQUERY"></script>
<script src="js/psynth.js"></script>

Quickstart

var graphParameters = {name: 'test map',
                       username: 'myusername',
                       password: 'mypassword',
                       url: 'https://psynth.psymphonic.com/'};
function MyFunction(graph){
     graph.addNode({name: "Sample Node"});
}
Psynth.createGraph(graphParameters, MyFunction);
